2.创建Girl.js

//构造函数
function Girl(option) {
    this._init(option);
}

//原型函数
Girl.prototype = {
  constructor:Girl,
    _init:function (option) {
        option = option || {};
        this.imgSrc = option.imgSrc;
        this.sWidth = option.sWidth || 256; //要裁剪的宽度
        this.sHeight = option.sHeight || 256; //要裁剪的高度
        this.x = option.x || 0;
        this.y = option.y || 0;
        this.width = option.width|| 256;
        this.height = option.height ||256;

        this.duration = option.duration || 200; //间隔执行时间
        this.dir = option.dir || 0;//方向
    }, //注意逗号

    //绘制函数
    render:function (ctx) {
        //备份指针
        var self = this;

        var image = new Image();
        image.src = this.imgSrc;
        image.onload = function () {
            //索引
            var index = 0;

            //定时器
            setInterval(function () {
                //清屏
                ctx.clearRect(self.x, self.y, self.width, self.height);
                //绘制
                ctx.drawImage(image, index * self.sWidth, self.dir *self.sHeight, self.sWidth, self.sHeight, self.x, self.y, self.width, self.height);
                //处理索引状态
                index++;
                index %= 8;
            }, self.duration);
        }
    },

    //改变方向
    changeDir:function (dir) {
        this.dir = dir;
    }

};
